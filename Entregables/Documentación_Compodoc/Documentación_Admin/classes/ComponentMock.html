    addComponent(c: MyComponent) {
        if (ComponentMock.comps.filter((e) => e.component_name == c.component_name).length > 0)
            return of({error: "error"});
        ComponentMock.comps.push(c);
        return of({next: "guardado"});
    }

    editComponent(c: MyComponent) {
        let editado = false;
        ComponentMock.comps.forEach((comp, index) => {
            if (comp.component_id == c.component_id) {
                ComponentMock.comps.splice(index, 1, c);
                editado = true;
            }
        });
        if (editado)
            return of({next: "guardado"});
        return of({error: "Error"});
    }

    deleteComponent(cId: number) {
        let borrado = false;
        ComponentMock.comps.forEach((comp, index) => {
            if (comp.component_id == cId) {
                ComponentMock.comps.splice(index, 1);
                borrado = true;
            }
        });
        if (borrado)
            return of({next: "borrado"});
        return of({error: "Error"});
    }
